export const emitFunctionName = (data: any) => {
  // 1. Broadcast para TODOS
  emitToAll('event:name', payload);
  
  // 2. Notificar destinatários específicos
  if (data.storeId) {
    emitToRoom(`store:${data.storeId}`, 'event:name', payload);
  }
  if (data.customerId) {  // ✅ NOVO PADRÃO
    emitToRoom(`user:${data.customerId}`, 'event:name', payload);
  }
  if (data.motoboyId) {   // ✅ NOVO PADRÃO
    emitToRoom(`user:${data.motoboyId}`, 'event:name', payload);
  }
};


Salas de Socket (Socket.io):
────────────────────────────

  user:{userId}        → Notificações pessoais (cliente/motoboy)
  store:{storeId}      → Notificações da loja
  motoboys             → Broadcast para todos motoboys


Eventos Emitidos:
─────────────────

  order:created              → Novo pedido criado
  order:updated              → Pedido atualizado
  order:status_changed       → Status do pedido mudou
  delivery:created           → Nova delivery criada
  delivery:updated           → Delivery atualizada
  delivery:status_changed    → Status da delivery mudou
  delivery:location_updated  → Localização do motoboy mudou
